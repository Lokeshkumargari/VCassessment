function DescendingArray(arr) {
  let n = arr.length;
  let swapped;
  do {
    swapped = false;
    for (let i = 0; i < n - 1; i++) {
      if (arr[i] < arr[i + 1]) {
        let temp = arr[i];
        arr[i] = arr[i + 1];
        arr[i + 1] = temp;
        swapped = true;
      }
    }
  } while (swapped);
  return arr;
}
const numbers = [2, 1, 9, 7, 3, 11];
const sortedNumbers = DescendingArray(numbers);
console.log(sortedNumbers);
